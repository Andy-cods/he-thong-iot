### 2.2 Pattern fix

#### P1 · Zod `.default(X)` output type undefined
Root: `z.coerce` hoặc `.default()` đã có trong schema nhưng TS `parse()` trả type input-strict.

**Fix:** cast qua variable trung gian bằng `z.output<typeof schema>`:

```ts
// Trước:
const parsed = listItemsSchema.parse(query);
return await listItems(parsed); // repo expect strict

// Sau (option A — clean):
type ListItemsInput = z.output<typeof listItemsSchema>;
const parsed: ListItemsInput = listItemsSchema.parse(query);
```

**Option B (simpler — nếu schema dùng `.default(X)` đã có giá trị):** sửa chữ ký repo nhận `Partial<X>` + merge default server-side. KHÔNG dùng — rối.

**Option C (nếu bug gốc là `.optional()` không có default):** thêm `.default()` vào zod schema:
```ts
// packages/shared/src/schemas/item.ts
export const listItemsSchema = z.object({
  sort: z.enum([...]).default("-updatedAt"),  // fix ở đây
  description: z.string().nullable().default(null),
  // ...
});
```

**Quyết định:** dùng Option C khi schema thiếu default; dùng Option A cast output khi không muốn đổi schema. Default = Option C cho query params (list).

### 3.4 Repo — 3 parallel count

**File:** `apps/web/src/server/repos/dashboard.ts`

```ts
import { sql } from "drizzle-orm";
import { db } from "@/lib/db";

export interface DashboardOverview {
  activeSkuCount: number;
  bomTemplateCount: number;
  supplierCount: number;
  lowStockCount: number | null;
  pendingPoCount: number | null;
  runningWoCount: number | null;
  cachedAt: string;
}

export async function getDashboardOverview(): Promise<DashboardOverview> {
  const [skuRow, bomRow, supRow] = await Promise.all([
    db.execute(sql`SELECT count(*)::int AS c FROM app.item WHERE is_active = true`),
    db.execute(sql`SELECT count(*)::int AS c FROM app.bom_template WHERE status = 'ACTIVE'`),
    db.execute(sql`SELECT count(*)::int AS c FROM app.supplier WHERE is_active = true`),
  ]);

  const activeSkuCount = Number((skuRow as any)[0]?.c ?? 0);
  const bomTemplateCount = Number((bomRow as any)[0]?.c ?? 0);
  const supplierCount = Number((supRow as any)[0]?.c ?? 0);

  return {
    activeSkuCount,
    bomTemplateCount,
    supplierCount,
    lowStockCount: null,    // V1.2 inventory module
    pendingPoCount: null,   // V1.2 PO module
    runningWoCount: null,   // V1.3 WO module
    cachedAt: new Date().toISOString(),
  };
}
```

### 3.5 Redis cache wrapper

**File:** `apps/web/src/server/services/dashboardCache.ts`

```ts
import IORedis from "ioredis";
import type { DashboardOverview } from "@/server/repos/dashboard";

const KEY = "dashboard:overview:v1";
const TTL_SECONDS = 60;

let client: IORedis | null = null;
function getClient() {
  if (!client) {
    client = new IORedis(process.env.REDIS_URL ?? "redis://localhost:6379/2", {
      lazyConnect: true,
      maxRetriesPerRequest: 2,
    });
  }
  return client;
}

export const dashboardCache = {
  async get(): Promise<DashboardOverview | null> {
    try {
      const raw = await getClient().get(KEY);
      return raw ? JSON.parse(raw) : null;
    } catch {
      return null;  // Redis down → cold path
    }
  },
  async set(data: DashboardOverview): Promise<void> {
    try {
      await getClient().set(KEY, JSON.stringify(data), "EX", TTL_SECONDS);
    } catch {
      // silent
    }
  },
  async invalidate(): Promise<void> {
    try {
      await getClient().del(KEY);
    } catch {}
  },
};
```

### 4.2 Fix race condition — worker

**File:** `apps/worker/src/jobs/bomImport.ts` line 243-256

**Trước:**
```ts
const [created] = await tx
  .insert(item)
  .values({
    sku: skuNorm,

    itemType: "PURCHASED",
    uom: "PCS",
    status: "ACTIVE",
    isActive: true,
    createdBy: input.actorId,
    updatedBy: input.actorId,
```

**Sau:**
```ts
const [upserted] = await tx
  .insert(item)
  .values({
    sku: skuNorm,
    name: nameFromDesc.slice(0, 250),
    itemType: "PURCHASED",
    uom: "PCS",
    status: "ACTIVE",
    isActive: true,
    createdBy: input.actorId,
    updatedBy: input.actorId,
  })
  .onConflictDoUpdate({
    target: item.sku,
    set: { updatedBy: input.actorId, updatedAt: new Date() },
  })
  .returning({ id: item.id });
```

**Mục tiêu:** `BomTreeView` hỗ trợ drop INTO (child) + drop BEFORE/AFTER (sibling) qua 3-zone detection, client guard cycle + depth, optimistic update parent+position (không level — chờ refetch).

### 5.2 Strategy — 3-zone drop detection

**Zone convention** (row height 32px):
- **TOP 30%** (0–9.6px từ top row) → drop làm SIBLING TRƯỚC over: `newParent = over.parentLineId`, `newPosition = over.position`
- **MIDDLE 40%** (9.6–22.4px) → drop làm CHILD của over: `newParent = over.id`, `newPosition = 1` (hoặc max+1 trong con của over)
- **BOTTOM 30%** (22.4–32px) → drop làm SIBLING SAU over: `newParent = over.parentLineId`, `newPosition = over.position + 1`

**Implementation:**
- @dnd-kit `DragMoveEvent` (`useDndMonitor` hoặc `onDragOver`) capture current pointer Y và over rect → set state `dropZone: "top" | "middle" | "bottom" | null`
- `handleDragEnd` đọc `dropZone` snapshot → compute params đúng

### 5.4 Refactor `handleDragEnd` + zone state

```tsx
// BomTreeView.tsx — thêm state
const [dropZone, setDropZone] = React.useState<"top" | "middle" | "bottom" | null>(null);
const [overId, setOverId] = React.useState<string | null>(null);

const handleDragOver = (event: DragOverEvent) => {
  const { over, activatorEvent, delta } = event;
  if (!over) {
    setDropZone(null);
    setOverId(null);
    return;
  }
  const rect = over.rect;
  const pointerY =
    (activatorEvent as PointerEvent).clientY + (delta?.y ?? 0);
  const relative = pointerY - rect.top;
  const ratio = relative / rect.height;
  const zone: "top" | "middle" | "bottom" =
    ratio < 0.3 ? "top" : ratio > 0.7 ? "bottom" : "middle";
  setDropZone(zone);
  setOverId(String(over.id));
};

const handleDragEnd = (event: DragEndEvent) => {
  const { active, over } = event;
  setDropZone(null);
  setOverId(null);

  if (!over || active.id === over.id) return;
  const activeNode = flat.find((n) => n.id === active.id);
  const overNode = flat.find((n) => n.id === over.id);
  if (!activeNode || !overNode) return;

  // Client guard cycle
  if (overNode.ancestorIds.includes(activeNode.id) || over.id === active.id) {
    toast.error("Không thể di chuyển vào con cháu của chính nó");
    return;
  }

  const zone = dropZone ?? "bottom";
  let newParent: string | null;
  let newPosition: number;
  let projectedLevel: number;

  if (zone === "middle") {
    // Drop INTO over → thành CHILD
    newParent = overNode.id;
    // newPosition = max(position) + 1 trong children của over
    newPosition =
      flat.filter((n) => n.parentLineId === overNode.id).length + 1;
    projectedLevel = overNode.level + 1;
  } else if (zone === "top") {
    newParent = overNode.parentLineId;
    newPosition = overNode.position;
    projectedLevel = overNode.level;
  } else {
    newParent = overNode.parentLineId;
    newPosition = overNode.position + 1;
    projectedLevel = overNode.level;
  }

  // Client guard depth: projected + subtree depth của active
  const activeSubtreeDepth = maxDepthBelow(flat, activeNode.id);
  if (projectedLevel + activeSubtreeDepth - 1 > BOM_MAX_LEVEL) {
    toast.error(`Vượt quá ${BOM_MAX_LEVEL} cấp BOM — thử drop ở vị trí nông hơn`);
    return;
  }

  onMove(String(active.id), newParent, newPosition);
};

// Helper
function maxDepthBelow(flat: BomFlatNode[], rootId: string): number {
  // Trả về số level sâu nhất dưới rootId (không bao gồm root)
  const rootLevel = flat.find((n) => n.id === rootId)?.level ?? 1;
  const descendants = flat.filter((n) =>
    n.ancestorIds.includes(rootId) || n.id === rootId
  );
  const maxLevel = Math.max(...descendants.map((n) => n.level), rootLevel);
  return maxLevel - rootLevel + 1;
}
```

```ts
"use client";
import * as React from "react";

export function useMediaQuery(query: string): boolean {
  const subscribe = React.useCallback(
    (cb: () => void) => {
      const mq = window.matchMedia(query);
      mq.addEventListener("change", cb);
      return () => mq.removeEventListener("change", cb);
    },
    [query],
  );
  const getSnapshot = () => window.matchMedia(query).matches;
  const getServerSnapshot = () => false;  // SSR default false

  return React.useSyncExternalStore(subscribe, getSnapshot, getServerSnapshot);
}

export const useIsMobile = () => useMediaQuery("(max-width: 767px)");
export const useIsTablet = () => useMediaQuery("(min-width: 768px) and (max-width: 1023px)");
```
