## §1. Tổng quan sprint V1.4

### 1.1 Mục tiêu cốt lõi

V1.4 biến hệ thống từ "chạy được" (V1.3) sang "production-grade" bằng cách đóng 3 vết thương còn hở:

- **Block Security (P0)** — 3 lớp guard bắt buộc đi **cùng nhau**, không tách được:
  - RLS (database layer — last line of defense)
  - RBAC 12×12 matrix (application layer — unified helper)
  - Rate limit (network layer — brute-force defense)
- **Block Operational (P1)** — 3 UI admin thực dụng:
  - Audit UI polish (diff viewer, export Excel, linked navigation)
  - Session management (revoke, force logout)
  - Reset password workflow (admin reset + stub self-reset)
- **Block Infra (P2)** — dưới gầm xe nhưng thiếu là đêm 3h VPS chết không ai biết:
  - Monitoring (Grafana Cloud Free tier — zero RAM VPS impact)
  - Backup cron R2 off-site + weekly restore drill
  - Load test k6 (máy local → VPS https://mes.songchau.vn)

`packages/shared/src/rbac/can.ts`:
```typescript
import type { Role } from "../types";
import { RBAC_MATRIX, type RbacAction, type RbacEntity } from "./matrix";

/**
 * can(['planner'], 'create', 'item') → true
 * can(['operator'], 'delete', 'item') → false
 *
 * Multi-role: OR — nếu có bất kỳ role thỏa mãn thì allow.
 */
export function can(
  roles: Role[] | null | undefined,
  action: RbacAction,
  entity: RbacEntity,
): boolean {
  if (!roles || roles.length === 0) return false;
  return roles.some((r) => RBAC_MATRIX[r]?.[entity]?.includes(action) ?? false);
}

/** Short-circuit: kiểm tra role có bất kỳ action nào trên entity (dùng cho nav menu). */
export function canAny(
  roles: Role[] | null | undefined,
  entity: RbacEntity,
```

```typescript
 * Wrap query trong transaction với SET LOCAL ROLE theo session.
 * Dùng trong mọi API route có session.
 *
 * Priority: admin > planner > warehouse > operator.
 */
export async function withRoleTx<T>(
  roles: Role[],
  fn: (tx: Transaction) => Promise<T>,
): Promise<T> {
  if (!env.RLS_ENABLED) {
    // Feature flag OFF → chạy thuần iot_app (backward compat V1.3)
    return db.transaction(fn);
  }

  const pgRole = pickHighestRole(roles); // 'app_admin' | 'app_planner' | ...

  return db.transaction(async (tx) => {
    await tx.execute(sql`SET LOCAL ROLE ${sql.raw(pgRole)}`);
    return fn(tx);
  });
}

function pickHighestRole(roles: Role[]): string {
  if (roles.includes("admin")) return "app_admin";
  if (roles.includes("planner")) return "app_planner";
  if (roles.includes("warehouse")) return "app_warehouse";
  if (roles.includes("operator")) return "app_operator";
  throw new Error("No valid role");
}
```

### 8.3 Task T-d-3: Rate limit middleware (1 ngày)

**File mới** — `apps/web/src/server/middlewares/rateLimit.ts`:
```typescript
import Redis from "ioredis";
import type { NextRequest } from "next/server";
import { NextResponse } from "next/server";
import { env } from "@/lib/env";

const redis = new Redis(env.REDIS_URL);

interface RateLimitOptions {
  bucket: string;      // "login" | "api" | "reset-pw"
  limit: number;       // requests
  windowSec: number;   // 60
}

/**
 * IP detection precedence:
 * 1. CF-Connecting-IP (Cloudflare Tunnel)
 * 2. X-Forwarded-For[0] (Caddy reverse proxy)
 * 3. req.ip (Next.js fallback)
 */
function getClientIp(req: NextRequest): string {
  return req.headers.get("cf-connecting-ip")
    ?? req.headers.get("x-forwarded-for")?.split(",")[0].trim()
    ?? req.ip
    ?? "0.0.0.0";
}

export async function rateLimit(
  req: NextRequest,
  opts: RateLimitOptions,
): Promise<NextResponse | null> {
  const ip = getClientIp(req);
  const key = `rl:${opts.bucket}:${ip}`;
  const now = Math.floor(Date.now() / 1000);

  // Fixed window: INCR + EXPIRE
  const pipe = redis.pipeline();
  pipe.incr(key);
  pipe.expire(key, opts.windowSec, "NX");   // chỉ set TTL lần đầu
  const [[, count]] = await pipe.exec() as [[null, number]];

  if (count > opts.limit) {
    // Log violation
    await db.insert(rateLimitLog).values({
      bucketKey: key,
      ipAddress: ip,
      userAgent: req.headers.get("user-agent") ?? null,
      requestPath: new URL(req.url).pathname,
    });

    const ttl = await redis.ttl(key);
    return NextResponse.json(
      { error: { code: "RATE_LIMITED", message: "Quá nhiều request. Thử lại sau." } },
      {
        status: 429,
        headers: {
          "Retry-After": String(ttl),
          "X-RateLimit-Limit": String(opts.limit),
          "X-RateLimit-Remaining": "0",
        },
      },
    );
  }

  return null;  // pass
}

// Helpers
export const loginRateLimit = (req: NextRequest) =>
  rateLimit(req, { bucket: "login", limit: 5, windowSec: 60 });
export const apiRateLimit = (req: NextRequest) =>
  rateLimit(req, { bucket: "api", limit: 60, windowSec: 60 });
export const resetPwRateLimit = (req: NextRequest) =>
  rateLimit(req, { bucket: "reset-pw", limit: 3, windowSec: 3600 });
```

**Apply trong route:**
```typescript

  const limited = await loginRateLimit(req);
  if (limited) return limited;

  // ... existing login logic
}

// apps/web/src/middleware.ts (global)
export async function middleware(req: NextRequest) {
  if (req.nextUrl.pathname.startsWith("/api/") &&
      !req.nextUrl.pathname.startsWith("/api/health") &&
      !req.nextUrl.pathname.startsWith("/api/ready")) {
    const limited = await apiRateLimit(req);
    if (limited) return limited;
  }
  return NextResponse.next();
}
```

### 10.1 Task T-f-1: Script update + R2 + retention (1 ngày)

**Script mới** — `deploy/scripts/backup-offsite.sh`:
```bash
#!/usr/bin/env bash
# Wrapper mới: pg_dump | gpg | rclone R2 + Telegram notify
set -euo pipefail

# Reuse config từ backup.sh
source /opt/hethong-iot/scripts/backup.sh.env   # ENV file

PG_CONTAINER="${PG_CONTAINER:-iot_postgres}"
PG_USER="${PG_USER:-hethong_app}"
PG_DB="${PG_DB:-hethong_iot}"
BACKUP_DIR="${BACKUP_DIR:-/opt/hethong-iot/backups}"
GPG_RECIPIENT="${GPG_RECIPIENT:-backup@iot.domain.vn}"
R2_REMOTE="${R2_REMOTE:-r2:iot-backups}"
TG_BOT="${TELEGRAM_BOT_TOKEN:-}"
TG_CHAT="${TELEGRAM_CHAT_ID:-}"

notify() {
  local msg="$1"
  echo "[backup $(date -Iseconds)] $msg"
  if [[ -n "$TG_BOT" && -n "$TG_CHAT" ]]; then
    curl -sS --max-time 10 \
      -d "chat_id=${TG_CHAT}" \
      -d "text=[iot backup] ${msg}" \
      "https://api.telegram.org/bot${TG_BOT}/sendMessage" > /dev/null || true
  fi
}
trap 'notify "FAIL dòng $LINENO (exit $?)"' ERR

TS="$(date +%Y%m%d-%H%M%S)"
DUMP_FILE="${BACKUP_DIR}/${PG_DB}-${TS}.sql.gz"
ENC_FILE="${DUMP_FILE}.gpg"

mkdir -p "$BACKUP_DIR"

# 1. pg_dump
notify "Bắt đầu pg_dump ${PG_DB}"
docker exec "$PG_CONTAINER" pg_dump -U "$PG_USER" -d "$PG_DB" -Fc --no-owner --no-privileges \
  | gzip -9 > "$DUMP_FILE"

# 2. gpg encrypt
gpg --batch --yes --encrypt --recipient "$GPG_RECIPIENT" --output "$ENC_FILE" "$DUMP_FILE"
rm -f "$DUMP_FILE"

SIZE=$(du -h "$ENC_FILE" | cut -f1)
notify "Dump OK: ${ENC_FILE} (${SIZE})"

# 3. Size sanity check (alert nếu bất thường)
SIZE_BYTES=$(stat -c %s "$ENC_FILE")
if [[ "$SIZE_BYTES" -lt 10485760 ]]; then  # < 10MB
  notify "WARN: Backup size ${SIZE_BYTES} bytes nhỏ bất thường"
fi

# 4. Upload R2
notify "Upload R2 → ${R2_REMOTE}/"
rclone copy "$ENC_FILE" "${R2_REMOTE}/" --progress --timeout=5m

# 5. Retention: local 7 ngày, R2 qua lifecycle rule (cài trong R2 console)
find "$BACKUP_DIR" -name "${PG_DB}-*.sql.gz.gpg" -mtime +7 -delete
notify "Xoá backup local > 7 ngày. Done."
```
