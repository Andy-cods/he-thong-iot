### 2.1.2 File `0003b_bom_template.sql` (app user)

```sql
-- ============================================================
-- engineering.bom_template — master BOM cho 1 product
-- ============================================================
CREATE TABLE engineering.bom_template (
    id              uuid PRIMARY KEY DEFAULT gen_random_uuid(),
    template_code   varchar(64) NOT NULL UNIQUE,              -- BOMT-<FGCODE>
    product_item_id uuid NOT NULL REFERENCES master.item(id) ON DELETE RESTRICT,
    name            varchar(255) NOT NULL,
    category        varchar(64),                               -- MACHINE|SPARE|JIG
    bom_type        varchar(16) NOT NULL DEFAULT 'MANUFACTURE',-- MANUFACTURE|SPARE
    owner_team      varchar(64),
    notes           text,
    is_active       boolean NOT NULL DEFAULT true,
    created_at      timestamptz NOT NULL DEFAULT now(),
    created_by      uuid,
    updated_at      timestamptz NOT NULL DEFAULT now(),
    updated_by      uuid,
    CONSTRAINT ck_bom_template_type CHECK (bom_type IN ('MANUFACTURE','SPARE'))
);
CREATE INDEX idx_bom_template_product ON engineering.bom_template (product_item_id);
CREATE INDEX idx_bom_template_name_trgm ON engineering.bom_template USING gin (name gin_trgm_ops);

-- ============================================================
-- engineering.bom_revision — version của template
-- ============================================================
CREATE TABLE engineering.bom_revision (
    id                uuid PRIMARY KEY DEFAULT gen_random_uuid(),
    template_id       uuid NOT NULL REFERENCES engineering.bom_template(id) ON DELETE CASCADE,
    revision_no       varchar(32) NOT NULL,                    -- R01, R02
    revision_status   varchar(16) NOT NULL DEFAULT 'DRAFT',    -- DRAFT, RELEASED, OBSOLETE
    effective_from    timestamptz,
    effective_to      timestamptz,
    released_at       timestamptz,
    released_by       uuid,
    notes             text,
    created_at        timestamptz NOT NULL DEFAULT now(),
    created_by        uuid,
    UNIQUE (template_id, revision_no),
    CONSTRAINT ck_bom_rev_status CHECK (
        revision_status IN ('DRAFT','RELEASED','OBSOLETE')
    ),
    CONSTRAINT ck_bom_rev_released CHECK (
        (revision_status = 'DRAFT' AND released_at IS NULL)
        OR (revision_status <> 'DRAFT' AND released_at IS NOT NULL)
    )
);
CREATE INDEX idx_bom_revision_template ON engineering.bom_revision (template_id, revision_no);
CREATE INDEX idx_bom_revision_status ON engineering.bom_revision (revision_status)
    WHERE revision_status = 'RELEASED';  -- Partial index cho query "latest released"

-- ============================================================
-- engineering.bom_component — line trong revision (tree parent-child)
-- ============================================================
CREATE TABLE engineering.bom_component (
    id                     uuid PRIMARY KEY DEFAULT gen_random_uuid(),
    revision_id            uuid NOT NULL REFERENCES engineering.bom_revision(id) ON DELETE CASCADE,
    parent_component_id    uuid REFERENCES engineering.bom_component(id) ON DELETE CASCADE,
    parent_item_id         uuid NOT NULL REFERENCES master.item(id),  -- redundant, cho query recursive nhanh
    child_item_id          uuid NOT NULL REFERENCES master.item(id),
    find_no                integer NOT NULL,                         -- thứ tự trong parent (line_no)
    qty_per_parent         numeric(18,6) NOT NULL CHECK (qty_per_parent > 0),
    scrap_pct              numeric(7,4) NOT NULL DEFAULT 0 CHECK (scrap_pct >= 0 AND scrap_pct < 1),
    source_type            varchar(16) NOT NULL DEFAULT 'BUY',       -- MAKE|BUY|PHANTOM
    phantom_flag           boolean NOT NULL DEFAULT false,
    substitute_group_code  varchar(64),                              -- NULL nếu không có alternate
    is_optional            boolean NOT NULL DEFAULT false,
    operation_seq          integer,
    qc_plan_code           varchar(64),
    drawing_ref            text,
    notes                  text,
    created_at             timestamptz NOT NULL DEFAULT now(),
    created_by             uuid,
    updated_at             timestamptz NOT NULL DEFAULT now(),
    CONSTRAINT ck_bom_comp_source CHECK (source_type IN ('MAKE','BUY','PHANTOM')),
    UNIQUE (revision_id, parent_component_id, find_no)
);
CREATE INDEX idx_bom_comp_revision ON engineering.bom_component (revision_id);
CREATE INDEX idx_bom_comp_parent ON engineering.bom_component (parent_component_id);
CREATE INDEX idx_bom_comp_parent_item ON engineering.bom_component (revision_id, parent_item_id);  -- covering cho recursive CTE
CREATE INDEX idx_bom_comp_child_item ON engineering.bom_component (child_item_id);
CREATE INDEX idx_bom_comp_substitute ON engineering.bom_component (substitute_group_code)
    WHERE substitute_group_code IS NOT NULL;

-- ============================================================
-- engineering.eco_request — change request stub (V1.1 workflow only, apply defer V1.3)
-- ============================================================
CREATE TABLE engineering.eco_request (
    id                      uuid PRIMARY KEY DEFAULT gen_random_uuid(),
    eco_no                  varchar(32) NOT NULL UNIQUE,              -- ECO-YYMM-####
    title                   varchar(255) NOT NULL,
    description             text,
    affected_template_id    uuid REFERENCES engineering.bom_template(id),
    affected_revision_id    uuid REFERENCES engineering.bom_revision(id),
    requester_user_id       uuid NOT NULL,
    approver_user_id        uuid,
    status                  varchar(16) NOT NULL DEFAULT 'DRAFT',
    submitted_at            timestamptz,
    approved_at             timestamptz,
    rejected_at             timestamptz,
    applied_at              timestamptz,                              -- V1.3 mới set
    reject_reason           text,
    created_at              timestamptz NOT NULL DEFAULT now(),
    CONSTRAINT ck_eco_status CHECK (
        status IN ('DRAFT','SUBMITTED','APPROVED','REJECTED','APPLIED')
    )
);
CREATE INDEX idx_eco_template ON engineering.eco_request (affected_template_id);
CREATE INDEX idx_eco_status ON engineering.eco_request (status);
```

    bom_template_id      uuid REFERENCES engineering.bom_template(id),  -- dùng khi item là product cần BOM

    item_description     varchar(255),
    qty                  numeric(18,6) NOT NULL CHECK (qty > 0),
    uom                  varchar(16) NOT NULL,
    unit_price           numeric(18,2),
    line_amount          numeric(18,2),
    due_date             date,
    notes                text,
    created_at           timestamptz NOT NULL DEFAULT now(),
    UNIQUE (order_id, line_no),
    CONSTRAINT ck_order_line_source CHECK (
        (bom_template_id IS NOT NULL AND item_id IS NULL)
        OR (bom_template_id IS NULL AND item_id IS NOT NULL)
    )
);
CREATE INDEX idx_order_line_order ON sales.customer_order_line (order_id);
CREATE INDEX idx_order_line_template ON sales.customer_order_line (bom_template_id);

-- ============================================================

-- ============================================================
CREATE TABLE sales.bom_snapshot (

    order_line_id            uuid NOT NULL REFERENCES sales.customer_order_line(id),
    snapshot_no              varchar(32) NOT NULL UNIQUE,               -- BOMS-<SO>-<seq>
    source_bom_template_id   uuid NOT NULL REFERENCES engineering.bom_template(id),

    root_qty                 numeric(18,6) NOT NULL,                    -- qty trên order line
    released_at              timestamptz NOT NULL DEFAULT now(),
    released_by              uuid,
    is_locked                boolean NOT NULL DEFAULT true,             -- TRUE sau release, không ai update được metadata
    notes                    text
);
CREATE INDEX idx_snapshot_order ON sales.bom_snapshot (order_id);
CREATE INDEX idx_snapshot_source ON sales.bom_snapshot (source_bom_revision_id);

-- Trigger: không cho UPDATE metadata snapshot sau khi lock
CREATE OR REPLACE FUNCTION sales.trg_snapshot_lock() RETURNS trigger AS $$
BEGIN
  IF OLD.is_locked = true AND (
    NEW.source_bom_revision_id <> OLD.source_bom_revision_id
    OR NEW.source_bom_template_id <> OLD.source_bom_template_id
    OR NEW.root_item_id <> OLD.root_item_id
    OR NEW.root_qty <> OLD.root_qty
  ) THEN
    RAISE EXCEPTION 'Snapshot % is locked, cannot modify metadata', OLD.snapshot_no;
  END IF;
  RETURN NEW;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER trg_snapshot_no_modify_locked
BEFORE UPDATE ON sales.bom_snapshot
FOR EACH ROW EXECUTE FUNCTION sales.trg_snapshot_lock();

## 3.6 Shortage Board endpoint

### `GET /api/shortage-board`
- **Auth:** `material_planner`, `purchasing`, `admin`.
- **Query:** `dueWithinDays` (default 30), `warehouseId` (V1.4 scope), `sortBy` (shortage_desc | due_asc).
- **SQL core:**
```sql
SELECT
  sl.item_id,
  i.item_code,
  i.name AS item_name,
  i.uom,
  SUM(sl.gross_required_qty) AS required_total,
  SUM(sl.qc_pass_qty + sl.reserved_qty + sl.issued_qty + sl.assembled_qty) AS covered,
  SUM(sl.remaining_short_qty) AS shortage,
  MIN(co.due_date) AS earliest_due_date,
  COUNT(DISTINCT sl.order_id) AS affected_orders
FROM sales.bom_snapshot_line sl
JOIN sales.customer_order co ON co.id = sl.order_id
JOIN master.item i ON i.id = sl.item_id
WHERE sl.line_status IN ('PLANNED','PURCHASING')
  AND sl.source_type = 'BUY'  -- V1.1 shortage chỉ cho hàng mua; MAKE sẽ V1.3
  AND sl.phantom_flag = false
  AND co.status IN ('CONFIRMED','IN_PRODUCTION')
  AND co.due_date <= CURRENT_DATE + ($1::int || ' days')::interval
GROUP BY sl.item_id, i.item_code, i.name, i.uom
HAVING SUM(sl.remaining_short_qty) > 0
ORDER BY earliest_due_date ASC, shortage DESC;
```

## 5.1 Definition 10 state

| From | To | Side effect | Guard condition |
|---|---|---|---|

## 5.3 Invalid transitions (80+ cases)

Ví dụ:
- PLANNED → AVAILABLE (skip PURCHASING + INBOUND_QC) → 422
- PLANNED → RESERVED (skip AVAILABLE) → 422
- AVAILABLE → ASSEMBLED (skip RESERVED, ISSUED) → 422
- ISSUED → PLANNED (reverse) → 422
- CLOSED → AVAILABLE → 422 (final state)

## 5.4 Code: `lib/state-machines/bom-snapshot-line.ts`

```ts
export type SnapshotLineStatus =
  | 'PLANNED' | 'PURCHASING' | 'IN_PRODUCTION'
  | 'INBOUND_QC' | 'PROD_QC' | 'AVAILABLE'
  | 'RESERVED' | 'ISSUED' | 'ASSEMBLED' | 'CLOSED';

// Allowed transition table (from -> to[])
const ALLOWED_TRANSITIONS: Record<SnapshotLineStatus, SnapshotLineStatus[]> = {
  PLANNED:       ['PURCHASING','IN_PRODUCTION','CLOSED'],
  PURCHASING:    ['INBOUND_QC','CLOSED','PLANNED'],
  IN_PRODUCTION: ['PROD_QC','CLOSED','PLANNED'],
  INBOUND_QC:    ['AVAILABLE','PLANNED'],
  PROD_QC:       ['AVAILABLE','IN_PRODUCTION'],
  AVAILABLE:     ['RESERVED','INBOUND_QC','CLOSED'],
  RESERVED:      ['AVAILABLE','ISSUED'],
  ISSUED:        ['ASSEMBLED','RESERVED'],
  ASSEMBLED:     ['CLOSED'],
  CLOSED:        [],
};

export function isValidTransition(
  from: SnapshotLineStatus,
  to: SnapshotLineStatus,
  userRole: string,
): { valid: boolean; reason?: string } {
  // Admin override (trừ CLOSED → *)
  if (userRole === 'admin' && from !== 'CLOSED') return { valid: true };

  const allowed = ALLOWED_TRANSITIONS[from] ?? [];
  if (!allowed.includes(to)) {
    return {
      valid: false,
      reason: `Không thể chuyển từ ${from} sang ${to}. Cho phép: [${allowed.join(', ')}]`,
    };
  }
  return { valid: true };
}

export async function transitionSnapshotLine(args: {
  lineId: string;
  toStatus: SnapshotLineStatus;
  actorUserId: string;
  actorRole: string;
  reason?: string;
  context?: Record<string, unknown>;
  db: Db;
}) {
  const { lineId, toStatus, actorUserId, actorRole, reason, context, db } = args;
  return await db.transaction(async (tx) => {
    const line = await tx.query.bomSnapshotLine.findFirst({ where: eq(bomSnapshotLine.id, lineId) });
    if (!line) throw new HttpError(404, 'Snapshot line not found');

    const check = isValidTransition(line.lineStatus, toStatus, actorRole);
    if (!check.valid) throw new HttpError(422, check.reason!);

    await tx.update(bomSnapshotLine)
      .set({
        lineStatus: toStatus,
        lastTransitionAt: new Date(),
        lastTransitionBy: actorUserId,
      })
      .where(eq(bomSnapshotLine.id, lineId));

    await tx.insert(stateTransitionLog).values({
      entityType: 'bom_snapshot_line',
      entityId: lineId,
      fromStatus: line.lineStatus,
      toStatus,
      actorUserId,
      reason,
      contextJson: context,
    });

    return { lineId, fromStatus: line.lineStatus, toStatus };
  });
}
```

# §6 Recursive CTE BOM explode

## 6.1 Bài toán

Input: `orderLineId`, `rootRevisionId`, `rootQty`.
Output: Insert vào `sales.bom_snapshot_line` toàn bộ component từ root đến leaf, kèm:
- `required_qty` = qty_per_parent × parent.required_qty (ancestor chain)
- `gross_required_qty` = required_qty × ∏ (1 + scrap_pct ancestor chain)
- `line_path` = "001/002/003" (tree sort)
- `depth` tăng dần
- Phantom component: skip explode **chính nó** (không insert) nhưng explode **children** với qty × phantom.qty_per_parent.

## 6.2 Core SQL

Chạy trong worker BullMQ `bom-explode.ts`, file `apps/worker/src/jobs/bom-explode.ts`:

```sql
-- Pseudo-code (Drizzle sql`` template)
INSERT INTO sales.bom_snapshot_line (
  id, snapshot_id, order_id,
  source_bom_component_id, parent_snapshot_line_id, parent_item_id,
  item_id, depth, line_path, find_no,
  source_type, phantom_flag, substitute_group_code,
  qty_per_parent, required_qty, gross_required_qty,
  scrap_pct_cumulative, uom,
  line_status, allocation_state, created_at
)
WITH RECURSIVE exploded AS (
  -- Anchor: root components của revision
  SELECT
    gen_random_uuid()                                   AS new_id,
    $1::uuid                                             AS snapshot_id,
    $2::uuid                                             AS order_id,
    c.id                                                 AS source_bom_component_id,
    NULL::uuid                                           AS parent_snapshot_line_id,
    NULL::uuid                                           AS parent_item_id,
    c.child_item_id                                      AS item_id,
    1                                                    AS depth,
    LPAD(c.find_no::text, 3, '0')                        AS line_path,
    c.find_no                                            AS find_no,
    c.source_type,
    c.phantom_flag,
    c.substitute_group_code,
    c.qty_per_parent                                     AS qty_per_parent,
    (c.qty_per_parent * $3::numeric)                     AS required_qty,
    (c.qty_per_parent * $3::numeric * (1 + c.scrap_pct)) AS gross_required_qty,
    c.scrap_pct                                          AS scrap_pct_cumulative,
    i.uom                                                AS uom
  FROM engineering.bom_component c
  JOIN master.item i ON i.id = c.child_item_id
  WHERE c.revision_id = $4::uuid
    AND c.parent_component_id IS NULL

  UNION ALL

  -- Recursive: children của current
  SELECT
    gen_random_uuid(),
    e.snapshot_id,
    e.order_id,
    c.id,
    e.new_id,
    e.item_id,                                           -- parent_item_id của child = item của ancestor
    c.child_item_id,
    e.depth + 1,
    e.line_path || '/' || LPAD(c.find_no::text, 3, '0'),
    c.find_no,
    c.source_type,
    c.phantom_flag,
    c.substitute_group_code,
    c.qty_per_parent,
    (e.required_qty * c.qty_per_parent),
    (e.gross_required_qty * c.qty_per_parent * (1 + c.scrap_pct)),
    (e.scrap_pct_cumulative + c.scrap_pct),  -- simple sum; more accurate: 1-(1-a)(1-b)
    i.uom
  FROM exploded e
  JOIN engineering.bom_component c ON c.parent_component_id = (
    SELECT bc.id FROM engineering.bom_component bc
    WHERE bc.revision_id = $4::uuid
      AND bc.id = e.source_bom_component_id
    LIMIT 1
  )
  JOIN master.item i ON i.id = c.child_item_id
  WHERE e.depth < 10  -- safety max depth
)
SELECT
  new_id, snapshot_id, order_id,
  source_bom_component_id, parent_snapshot_line_id, parent_item_id,
  item_id, depth, line_path, find_no,
  source_type, phantom_flag, substitute_group_code,
  qty_per_parent, required_qty, gross_required_qty,
  scrap_pct_cumulative, uom,
  'PLANNED' AS line_status,
  'UNALLOCATED' AS allocation_state,
  now() AS created_at
FROM exploded
WHERE phantom_flag = false  -- phantom không tạo row, nhưng children vẫn explode ra
ORDER BY line_path;
```

**Params:** `$1=snapshotId`, `$2=orderId`, `$3=rootQty`, `$4=revisionId`.

## 6.3 Phantom handling

- Phantom component trong `bom_component`: `phantom_flag=true`.
- Recursive CTE **duyệt qua** phantom (để đi xuống children), nhưng clause `WHERE phantom_flag = false` lọc ở insert step.
- Children của phantom vẫn được insert với `parent_snapshot_line_id` = snapshot line của ancestor **gần nhất không phải phantom**.

**V1.1 simplification:** Assume phantom chỉ 1 cấp. Triple-nested phantom defer V1.2 (xem §13).

## 6.4 Substitute handling

- V1.1: Explode **primary item** của substitute_group (item đầu tiên theo `find_no` trong cùng parent + cùng substitute_group_code).
- `substitute_group_code` giữ trong snapshot_line để V1.2 reservation có thể chọn alternate theo stock.
- Not in V1.1: UI picker cho substitute (defer V1.3).

## 6.5 Scrap % cumulative

- Mỗi component có `scrap_pct` (0-0.9999).
- Gross required chain: `gross = required × ∏ (1 + scrap_pct_ancestor)`.
- V1.1 dùng **simple multiplication** (đủ chính xác cho xưởng cơ khí). V1.2 nếu cần probability compounding sẽ upgrade.

## 6.6 Deterministic ordering

- `ORDER BY line_path` đảm bảo thứ tự explode = thứ tự user thấy trong editor.
- `line_path` format `LPAD(findNo, 3, '0')` chuỗi dot-separated → sort string đúng numeric.

      const revId = await seedDeepBom({ depth: 4, childrenPerLevel: Math.ceil(n ** 0.25) });
      const start = Date.now();
      await explodeBom({ revisionId: revId, rootQty: 1, snapshotId, orderId });
      const elapsed = Date.now() - start;
      console.log(`n=${n}: ${elapsed}ms`);
      if (n === 1000) expect(elapsed).toBeLessThan(5000);
    });
  }
});
```

**Fallback khi > 5s:**
- Chunk insert 500 rows/batch (tránh Postgres insert timeout).
- Async BullMQ với polling UI 2s (reuse pattern Import Wizard V1.0).
- Nếu vẫn chậm > 10s: flag `LARGE_BOM_ASYNC=true` → chạy background, UI show toast "Đang explode...".

## 7.1 Định nghĩa 9 cột

| # | Column | VN | Công thức/Ý nghĩa |
|---|---|---|---|

| 3 | `open_purchase_qty` | SL đã đặt mua | ∑ PO line linked_snapshot_line_id = this (V1.2) |
| 4 | `received_qty` | SL đã nhận | ∑ GRN line qty (V1.2) |

| 6 | `reserved_qty` | SL đã reserve | ∑ stock_reservation qty còn active (V1.2) |

| 8 | `assembled_qty` | SL đã lắp vào FG | ∑ assembly_scan COMPLETE (V1.3) |
| 9 | `remaining_short_qty` | SL còn thiếu | **Generated: GREATEST(gross_required - qc_pass - reserved - issued - assembled, 0)** |

## 7.2 Invariants (check trong unit test + DB constraint)

**I1:** `gross_required_qty >= required_qty` (scrap cộng vào luôn dương).
**I2:** `received_qty >= qc_pass_qty` (QC chỉ pass được phần đã nhận).
**I3:** `qc_pass_qty >= reserved_qty + issued_qty + assembled_qty` (chỉ reserve/issue hàng đã pass QC).
**I4:** `issued_qty >= assembled_qty` (lắp chỉ được phần đã issue).
**I5:** `remaining_short_qty >= 0` (không âm).
**I6:** Khi status = CLOSED: `assembled_qty + scrapped_qty = gross_required_qty` (V1.3 sẽ thêm scrapped_qty).

export async function getDashboardOverview({ db, userId, userRole }: Ctx) {
  // Cache Redis 30s
  const cached = await redis.get('dashboard:overview');
  if (cached) return JSON.parse(cached);

  const [activeItems, shortageCount, ordersReadiness, alerts] = await Promise.all([
    db.select({ c: count() }).from(item).where(and(eq(item.isActive, true), isNull(item.deletedAt))),
    db.execute(sql`
      SELECT COUNT(DISTINCT item_id) AS c
      FROM sales.bom_snapshot_line sl
      JOIN sales.customer_order co ON co.id = sl.order_id
      WHERE sl.line_status IN ('PLANNED','PURCHASING')
        AND sl.remaining_short_qty > 0
        AND co.status IN ('CONFIRMED','IN_PRODUCTION')
    `),
    getOrdersReadinessTop10(db),
    getAlertsFromShortage(db),
  ]);

  const result = {
    kpis: {
      activeItems: activeItems[0].c,
      pendingPo: 0,      // TODO V1.2
      runningWo: 0,      // TODO V1.3
      lowStockAlerts: shortageCount.rows[0].c,
    },
    ordersReadiness,
    alerts,
    systemHealth: await getSystemHealth(),
  };

  await redis.setex('dashboard:overview', 30, JSON.stringify(result));
  return result;
}

async function getOrdersReadinessTop10(db) {
  // Top 10 orders earliest due, compute readiness%
  return db.execute(sql`
    WITH line_progress AS (
      SELECT
        order_id,
        COUNT(*) AS total_lines,
        COUNT(*) FILTER (WHERE line_status IN ('AVAILABLE','RESERVED','ISSUED','ASSEMBLED','CLOSED')) AS ready_lines
      FROM sales.bom_snapshot_line
      GROUP BY order_id
    )
    SELECT co.id AS order_id, co.order_no, co.customer_name, co.due_date,
           lp.total_lines, lp.ready_lines,
           CASE WHEN lp.total_lines > 0 THEN ROUND(lp.ready_lines::numeric / lp.total_lines * 100, 1) ELSE 0 END AS readiness_pct
    FROM sales.customer_order co
    LEFT JOIN line_progress lp ON lp.order_id = co.id
    WHERE co.status IN ('CONFIRMED','IN_PRODUCTION')
    ORDER BY co.due_date ASC, co.priority ASC
    LIMIT 10
  `).then(r => r.rows);
}

async function getAlertsFromShortage(db) {
  // Shortage critical = shortage > 0 AND due < 7d
  return db.execute(sql`
    SELECT
      'critical' AS severity,
      'shortage' AS type,
      i.name AS title,
      'Thiếu ' || SUM(sl.remaining_short_qty) || ' ' || i.uom ||
        ', due ' || MIN(co.due_date) AS detail,
      'item' AS entity_type,
      sl.item_id AS entity_id
    FROM sales.bom_snapshot_line sl
    JOIN sales.customer_order co ON co.id = sl.order_id
    JOIN master.item i ON i.id = sl.item_id
    WHERE sl.line_status IN ('PLANNED','PURCHASING')
      AND sl.remaining_short_qty > 0
      AND co.status IN ('CONFIRMED','IN_PRODUCTION')
      AND co.due_date <= CURRENT_DATE + INTERVAL '7 days'
    GROUP BY sl.item_id, i.name, i.uom
    ORDER BY MIN(co.due_date) ASC
    LIMIT 5
  `).then(r => r.rows);
}
```
