COMPLETED   → (final, no transition)
CANCELLED   → (final, no transition)

**Guard `releaseWorkOrder`:**

```typescript
// Snippet (đầy đủ trong §16)
export async function reserveSnapshotLine(input: {
  snapshotLineId: string;
  qty: number;
  policy?: "AUTO" | "FIFO" | "FEFO";
  userId: string;
  woId?: string;
}): Promise<{ reservationId: string; lotId: string; lotCode: string; pickedQty: number }> {
  return await db.transaction(async (tx) => {
    // 1. Set lock timeout
    await tx.execute(sql`SET LOCAL lock_timeout = '5s'`);

    // 2. Query snapshot_line + item
    const line = await tx.select({
      id: bomSnapshotLine.id,
      itemId: bomSnapshotLine.componentItemId,
      requiredQty: bomSnapshotLine.grossRequiredQty,
      reservedQty: bomSnapshotLine.reservedQty,
      state: bomSnapshotLine.state,
      expiryDays: item.expiryDays, // JOIN item
    })...;

    // 3. Validate state
    if (!["AVAILABLE", "PURCHASING", "INBOUND_QC"].includes(line.state)) {
      throw new HttpError(400, `State ${line.state} không cho phép reserve`);
    }

    // 4. Advisory lock per item_id
    await tx.execute(sql`SELECT app.lock_item_reservation(${line.itemId})`);

    // 5. Determine policy
    const effectivePolicy = input.policy === "AUTO" || !input.policy
      ? (Number(line.expiryDays) > 0 ? "FEFO" : "FIFO")
      : input.policy;

    // 6. Query available lots + compute available_qty (on_hand - active_reservation_sum)
    const lots = await pickLotsWithPolicy(tx, line.itemId, effectivePolicy, input.qty);
    if (lots.length === 0 || sumQty(lots) < input.qty) {
      throw new HttpError(409, `Không đủ tồn kho: cần ${input.qty}, có ${sumQty(lots)}`);
    }

    // 7. Insert reservation (có thể nhiều lot nếu 1 lot không đủ)
    const reservations = [];
    for (const { lot, pickQty } of lots) {
      const [res] = await tx.insert(reservation).values({
        snapshotLineId: input.snapshotLineId,
        lotSerialId: lot.id,
        woId: input.woId,
        reservedQty: pickQty,
        reservationReason: effectivePolicy === "FIFO" ? "AUTO_FIFO" : "AUTO_FEFO",
        reservedBy: input.userId,
      }).returning();
      reservations.push(res);

      // 8. Insert inv_txn RESERVE
      await tx.insert(inventoryTxn).values({
        txType: "RESERVE",
        itemId: line.itemId,
        qty: pickQty,
        lotSerialId: lot.id,
        refTable: "reservation",
        refId: res.id,
        postedBy: input.userId,
      });
    }

    // 9. Update snapshot_line.reserved_qty + transition to RESERVED if full
    const newReservedQty = Number(line.reservedQty) + input.qty;
    await tx.update(bomSnapshotLine)
      .set({
        reservedQty: newReservedQty.toString(),
        state: newReservedQty >= Number(line.requiredQty) ? "RESERVED" : line.state,
        transitionedAt: new Date(),
        transitionedBy: input.userId,
        versionLock: sql`${bomSnapshotLine.versionLock} + 1`,
      })
      .where(eq(bomSnapshotLine.id, input.snapshotLineId));

    // 10. Audit RESERVE
    await tx.insert(auditLog).values({
      action: "RESERVE", entityType: "bom_snapshot_line", entityId: input.snapshotLineId,
      userId: input.userId, context: { lots: lots.map(l => ({id: l.lot.id, qty: l.pickQty})), policy: effectivePolicy },
    });

    return reservations[0]; // return primary
  });
}

async function pickLotsWithPolicy(
  tx: DrizzleTx, itemId: string, policy: "FIFO" | "FEFO", qtyNeeded: number
): Promise<{lot: InventoryLotSerial; pickQty: number}[]> {
  // Query lots AVAILABLE với on_hand = sum(inv_txn) - sum(active_reservation)
  const orderByClause = policy === "FEFO"
    ? sql`COALESCE(exp_date, '9999-12-31'::date) ASC, created_at ASC`
    : sql`created_at ASC`;

  // Filter: exclude lot < 7 day exp (EXPIRING_SOON)
  const lots = await tx.execute(sql`
    SELECT l.id, l.lot_code, l.exp_date, l.created_at,
           COALESCE(SUM(t.qty) FILTER (WHERE t.tx_type = 'IN_RECEIPT'), 0)
           - COALESCE(SUM(t.qty) FILTER (WHERE t.tx_type IN ('OUT_ISSUE','ASSEMBLY_CONSUME')), 0)
           - COALESCE((SELECT SUM(reserved_qty) FROM app.reservation r
                       WHERE r.lot_serial_id = l.id AND r.released_at IS NULL), 0) AS available_qty
    FROM app.inventory_lot_serial l
    LEFT JOIN app.inventory_txn t ON t.lot_serial_id = l.id
    WHERE l.item_id = ${itemId}
      AND l.status = 'AVAILABLE'
      AND (l.exp_date IS NULL OR l.exp_date > CURRENT_DATE + INTERVAL '7 days')
    GROUP BY l.id
    HAVING ... > 0
    ORDER BY ${orderByClause}
  `);

  // Greedy pick: cumulative sum until satisfied
  const picks = [];
  let remaining = qtyNeeded;
  for (const lot of lots) {
    if (remaining <= 0) break;
    const pick = Math.min(Number(lot.available_qty), remaining);
    picks.push({ lot, pickQty: pick });
    remaining -= pick;
  }
  return picks;
}
```

export async function createAssemblyScan(input: {
  woId: string;
  snapshotLineId: string;
  lotSerialId: string;
  qty: number;
  offlineQueueId: string;  // UUIDv7, idempotency key
  userId: string;
  deviceId?: string;
  scannedAt: Date;
}): Promise<{ scanId: string; alreadyExists: boolean }> {
  return await db.transaction(async (tx) => {
    // 1. Idempotency check
    const existing = await tx.select().from(assemblyScan)
      .where(eq(assemblyScan.offlineQueueId, input.offlineQueueId)).limit(1);
    if (existing.length) {
      return { scanId: existing[0].id, alreadyExists: true };
    }

    // 2. Advisory lock per lot (prevent duplicate consume)
    await tx.execute(sql`SET LOCAL lock_timeout = '5s'`);
    await tx.execute(sql`SELECT pg_advisory_xact_lock(hashtext('scan:' || ${input.lotSerialId}::text))`);

    // 3. Verify reservation active cho (snapshot_line, lot)
    const [res] = await tx.select().from(reservation)
      .where(and(
        eq(reservation.snapshotLineId, input.snapshotLineId),
        eq(reservation.lotSerialId, input.lotSerialId),
        isNull(reservation.releasedAt),
      ));
    if (!res) throw new HttpError(400, "LOT_NOT_RESERVED_FOR_LINE");
    if (Number(res.reservedQty) < input.qty) {
      throw new HttpError(400, `RESERVED_QTY_INSUFFICIENT: reserved=${res.reservedQty} scan=${input.qty}`);
    }

    // 4. Insert assembly_scan
    const [scan] = await tx.insert(assemblyScan).values({
      aoId: null, // defer assembly_order 1:1
      woId: input.woId,
      snapshotLineId: input.snapshotLineId,
      lotSerialId: input.lotSerialId,
      reservationId: res.id,
      offlineQueueId: input.offlineQueueId,
      scanKind: "CONSUME",
      barcode: "(from lot)", // enrich from lot lookup
      itemId: null,
      qty: input.qty.toString(),
      scannedAt: input.scannedAt,
      scannedBy: input.userId,
      deviceId: input.deviceId,
    }).returning();

    // 5. Insert inv_txn ASSEMBLY_CONSUME (negative qty)
    await tx.insert(inventoryTxn).values({
      txType: "ASSEMBLY_CONSUME",
      itemId: res.itemId, // lookup via lot
      qty: (-input.qty).toString(),
      lotSerialId: input.lotSerialId,
      refTable: "assembly_scan",
      refId: scan.id,
      postedBy: input.userId,
    });

    // 6. Update reservation (partial release hoặc full release)
    const newReservedQty = Number(res.reservedQty) - input.qty;
    if (newReservedQty <= 0.0001) {
      await tx.update(reservation)
        .set({ releasedAt: new Date(), releasedBy: input.userId, releaseReason: "CONSUMED_BY_SCAN" })
        .where(eq(reservation.id, res.id));
    } else {
      await tx.update(reservation)
        .set({ reservedQty: newReservedQty.toString() })
        .where(eq(reservation.id, res.id));
    }

    // 7. Update snapshot_line qty + state transition
    const [line] = await tx.select().from(bomSnapshotLine)
      .where(eq(bomSnapshotLine.id, input.snapshotLineId));
    const newIssued = Number(line.issuedQty) + input.qty;
    const newAssembled = Number(line.assembledQty) + input.qty; // V1.3 issue = assemble
    const required = Number(line.requiredQty);
    await tx.update(bomSnapshotLine)
      .set({
        issuedQty: newIssued.toString(),
        assembledQty: newAssembled.toString(),
        reservedQty: (Number(line.reservedQty) - input.qty).toString(),
        state: newAssembled >= required ? "ASSEMBLED" : "ISSUED",
        transitionedAt: new Date(),
        transitionedBy: input.userId,
        versionLock: sql`${bomSnapshotLine.versionLock} + 1`,
      })
      .where(eq(bomSnapshotLine.id, input.snapshotLineId));

    // 8. Update work_order_line.issued_qty + assembled_qty
    await tx.update(workOrderLine)
      .set({
        issuedQty: sql`${workOrderLine.issuedQty} + ${input.qty}`,
        assembledQty: sql`${workOrderLine.assembledQty} + ${input.qty}`,
      })
      .where(and(
        eq(workOrderLine.woId, input.woId),
        eq(workOrderLine.snapshotLineId, input.snapshotLineId),
      ));

    // 9. Update lot status if on_hand = 0
    const [lotOnHand] = await tx.execute(sql`
      SELECT COALESCE(SUM(qty), 0) AS on_hand FROM app.inventory_txn WHERE lot_serial_id = ${input.lotSerialId}
    `);
    if (Number(lotOnHand.on_hand) <= 0.0001) {
      await tx.update(inventoryLotSerial)
        .set({ status: "CONSUMED" })
        .where(eq(inventoryLotSerial.id, input.lotSerialId));
    }

    // 10. Audit
    await tx.insert(auditLog).values({
      action: "ISSUE", entityType: "assembly_scan", entityId: scan.id,
      userId: input.userId, context: { woId: input.woId, lotId: input.lotSerialId, qty: input.qty },
    });

    return { scanId: scan.id, alreadyExists: false };
  });
}

export async function getLotTimeline(lotId: string): Promise<LotTimeline> {
  const lot = await db.select().from(inventoryLotSerial)
    .where(eq(inventoryLotSerial.id, lotId)).then(r => r[0]);
  if (!lot) throw new HttpError(404, "LOT_NOT_FOUND");

  // 1. Receipt event (tìm inv_txn IN_RECEIPT đầu tiên + receiving_event link)
  const receipts = await db.select({
    txId: inventoryTxn.id,
    qty: inventoryTxn.qty,
    occurredAt: inventoryTxn.occurredAt,
    refId: inventoryTxn.refId,
    refTable: inventoryTxn.refTable,
    postedBy: inventoryTxn.postedBy,
  }).from(inventoryTxn)
    .where(and(eq(inventoryTxn.lotSerialId, lotId), eq(inventoryTxn.txType, "IN_RECEIPT")));

  // 2. Reservation history (active + historical)
  const reservations = await db.select({
    resId: reservation.id, resQty: reservation.reservedQty,
    reservedAt: reservation.reservedAt, releasedAt: reservation.releasedAt,
    reason: reservation.reservationReason, snapshotLineId: reservation.snapshotLineId,
    woId: reservation.woId,
  }).from(reservation).where(eq(reservation.lotSerialId, lotId));

  // 3. Assembly scan (issued)
  const scans = await db.select({
    scanId: assemblyScan.id, qty: assemblyScan.qty, scannedAt: assemblyScan.scannedAt,
    woId: assemblyScan.woId, scannedBy: assemblyScan.scannedBy,
  }).from(assemblyScan).where(eq(assemblyScan.lotSerialId, lotId));

  // 4. QC status changes (from audit log)
  const qcEvents = await db.select().from(auditLog)
    .where(and(eq(auditLog.entityType, "inventory_lot_serial"), eq(auditLog.entityId, lotId)));

  // 5. Current on_hand
  const [onHand] = await db.execute(sql`
    SELECT COALESCE(SUM(qty), 0) AS on_hand FROM app.inventory_txn WHERE lot_serial_id = ${lotId}
  `);

  // 6. Merge tất cả event theo timestamp
  const events = [
    ...receipts.map(r => ({ kind: "RECEIVED", at: r.occurredAt, data: r })),
    ...reservations.flatMap(r => [
      { kind: "RESERVED", at: r.reservedAt, data: r },
      r.releasedAt ? { kind: "RELEASED", at: r.releasedAt, data: r } : null,
    ].filter(Boolean)),
    ...scans.map(s => ({ kind: "CONSUMED", at: s.scannedAt, data: s })),
    ...qcEvents.map(q => ({ kind: "QC_CHANGE", at: q.createdAt, data: q })),
  ].sort((a, b) => a.at.getTime() - b.at.getTime());

  return { lot, events, onHand: Number(onHand.on_hand) };
}
