```sql
-- Materialized view shortage
CREATE MATERIALIZED VIEW app.mv_shortage_by_item AS
SELECT
  bsl.child_item_id                AS item_id,
  i.sku                            AS sku,
  i.name                           AS name,
  SUM(bsl.gross_required_qty)      AS total_required,
  SUM(bsl.qc_pass_qty)             AS total_available_alloc,
  SUM(bsl.open_purchase_qty)       AS total_on_order,
  SUM(bsl.remaining_short_qty)     AS total_short,
  COUNT(DISTINCT bsl.snapshot_id)  AS order_count,
  MAX(so.due_date)                 AS latest_due_date
FROM app.bom_snapshot_line bsl
JOIN app.bom_snapshot bs ON bs.id = bsl.snapshot_id
JOIN app.sales_order so ON so.id = bs.order_id
JOIN app.item i ON i.id = bsl.child_item_id
WHERE bsl.line_status IN ('PLANNED','PURCHASING','INBOUND_QC')
  AND so.status NOT IN ('CANCELLED','CLOSED')
  AND bsl.remaining_short_qty > 0
GROUP BY bsl.child_item_id, i.sku, i.name
HAVING SUM(bsl.remaining_short_qty) > 0;
```

```typescript
  // 1. Query bom_line tree của template
  const lines = await tx.select().from(bomLine).where(eq(bomLine.templateId, templateId));
  // 2. Compute tree_hash (canonical JSON SHA-256)
  const hash = computeTreeHash(lines);
  // 3. Check duplicate

    .where(and(eq(bomRevision.templateId, templateId), eq(bomRevision.treeHash, hash)));
  if (dup.length) throw new HttpError(409, "Không có thay đổi so với revision trước");
  // 4. Sinh revision_no (REL{nextSeq})
  const maxRev = await tx.execute(sql`SELECT COUNT(*) FROM app.bom_revision WHERE template_id = ${templateId}`);
  const revNo = `REL${String(Number(maxRev[0].count) + 1).padStart(2, "0")}`;
```

```typescript
export function computeTreeHash(lines: BomLine[]): string {
  const canonical = lines
    .map(l => ({ parent: l.parentLineId, component: l.componentItemId, level: l.level, position: l.position, qty: Number(l.qtyPerParent), scrap: Number(l.scrapPercent), uom: l.uom ?? "" }))
    .sort((a,b) => a.level - b.level || a.position - b.position);
  return crypto.createHash("sha256").update(JSON.stringify(canonical)).digest("hex");
}
```

```typescript
export const SNAPSHOT_EXPLODE_CTE = sql`
WITH RECURSIVE exploded AS (
  SELECT
    gen_random_uuid() AS new_id,
    ${sql.placeholder("snapshotId")}::uuid AS snapshot_id,
    ${sql.placeholder("orderId")}::uuid AS order_id,
    brl.id AS source_bom_line_id,
    NULL::uuid AS parent_snapshot_line_id,
    NULL::uuid AS parent_item_id,
    brl.component_item_id AS child_item_id,
    1 AS level,
    1 AS depth,
    LPAD(brl.position::text, 3, '0') AS line_path,
    brl.position AS find_no,
    brl.position AS position_no,
    COALESCE(i.default_source_type, 'BUY') AS source_type,
    brl.phantom_flag,
    brl.substitute_group_code,
    brl.qty_per_parent AS qty_per,
    brl.scrap_percent AS scrap_pct,
    (brl.qty_per_parent * ${sql.placeholder("rootQty")}::numeric) AS required_qty,
    (brl.qty_per_parent * ${sql.placeholder("rootQty")}::numeric * (1 + brl.scrap_percent)) AS gross_required_qty
  FROM app.bom_revision_line brl
  JOIN app.item i ON i.id = brl.component_item_id
  WHERE brl.revision_id = ${sql.placeholder("revisionId")}::uuid
    AND brl.parent_revision_line_id IS NULL
  UNION ALL
  SELECT
    gen_random_uuid(), e.snapshot_id, e.order_id,
    child.id, e.new_id, e.child_item_id, child.component_item_id,
    e.level + 1, e.depth + 1,
    e.line_path || '/' || LPAD(child.position::text, 3, '0'),
    child.position, child.position,
    COALESCE(i.default_source_type, 'BUY'),
    child.phantom_flag, child.substitute_group_code,
    child.qty_per_parent, child.scrap_percent,
    (e.required_qty * child.qty_per_parent),
    (e.gross_required_qty * child.qty_per_parent * (1 + child.scrap_percent))
  FROM exploded e
  JOIN app.bom_revision_line child ON child.parent_revision_line_id = e.source_bom_line_id
  JOIN app.item i ON i.id = child.component_item_id
  WHERE e.depth < 5
)
INSERT INTO app.bom_snapshot_line (
  id, snapshot_id, order_id, source_bom_line_id, parent_snapshot_line_id,
  parent_item_id, child_item_id, level, depth, line_path, find_no, position_no,
  source_type, phantom_flag, substitute_group_code,
  qty_per, scrap_pct, required_qty, gross_required_qty,
  line_status, frozen_at
)
SELECT
  new_id, snapshot_id, order_id, source_bom_line_id, parent_snapshot_line_id,
  COALESCE(parent_item_id, ${sql.placeholder("rootItemId")}::uuid),
  child_item_id, level, depth, line_path, find_no, position_no,
  source_type, phantom_flag, substitute_group_code,
  qty_per, scrap_pct, required_qty, gross_required_qty,
  'PLANNED', now()
FROM exploded
WHERE phantom_flag = false
ORDER BY line_path
RETURNING id;
`;
```

**Function `explodeSnapshot`:**
```typescript
export async function explodeSnapshot(
  orderId: string, revisionId: string, userId: string
): Promise<{ snapshotId: string; lineCount: number; explodeMs: number; async: boolean }> {
  const order = await getOrder(orderId);
  const revision = await getRevision(revisionId);
  if (revision.status !== "RELEASED") throw new HttpError(400, "Revision chưa RELEASED");
  
  // Estimate line count
  const [{ count }] = await db.execute(sql`SELECT COUNT(*) FROM app.bom_revision_line WHERE revision_id = ${revisionId}`);
  const estimate = Number(count) * 1.5; // branching multiplier rough
  
  if (estimate > Number(process.env.LARGE_BOM_ASYNC_THRESHOLD ?? 500)) {
    // Queue BullMQ job
    const jobId = await snapshotQueue.add("explode", { orderId, revisionId, userId });
    return { snapshotId: "", lineCount: 0, explodeMs: 0, async: true };
  }
  
  const start = Date.now();
  const snapshotCode = `SNP-${format(new Date(), "yyMMdd")}-${nanoid(4).toUpperCase()}`;
  
  return await db.transaction(async (tx) => {
    // 1. Insert header
    const [snap] = await tx.insert(bomSnapshot).values({
      snapshotCode,
      orderId,
      revisionId,
      rootItemId: order.productItemId,
      rootQty: order.orderQty,
      createdBy: userId,
    }).returning();
    
    // 2. Run CTE
    const inserted = await tx.execute(sql.raw(SNAPSHOT_EXPLODE_CTE, {
      snapshotId: snap.id,
      orderId,
      rootQty: Number(order.orderQty),
      revisionId,
      rootItemId: order.productItemId,
    }));
    
    const lineCount = inserted.length;
    const explodeMs = Date.now() - start;
    
    // 3. Update header telemetry
    await tx.update(bomSnapshot)
      .set({ totalLines: lineCount, explodeMs, maxDepth: Math.max(...inserted.map(r => r.depth ?? 1)) })
      .where(eq(bomSnapshot.id, snap.id));
    
    // 4. Update sales_order.snapshot_at/by + status → SNAPSHOTTED
    await tx.update(salesOrder)
      .set({ snapshotAt: new Date(), snapshotBy: userId, status: "SNAPSHOTTED" })
      .where(eq(salesOrder.id, orderId));
    
    // 5. Audit SNAPSHOT
    await tx.insert(auditLog).values({ action: "SNAPSHOT", entityType: "bom_snapshot", entityId: snap.id, userId, context: { orderId, revisionId, lineCount, explodeMs } });
```

```typescript
export const BOM_SNAPSHOT_LINE_STATES = [
  "PLANNED","PURCHASING","IN_PRODUCTION","INBOUND_QC","PROD_QC",
  "AVAILABLE","RESERVED","ISSUED","ASSEMBLED","CLOSED",
] as const;

export type BomSnapshotLineState = typeof BOM_SNAPSHOT_LINE_STATES[number];

// 20 valid transitions (80 invalid)
export const TRANSITION_MATRIX: Record<BomSnapshotLineState, BomSnapshotLineState[]> = {
  PLANNED: ["PURCHASING", "IN_PRODUCTION", "AVAILABLE", "CLOSED"], // direct AVAILABLE nếu item có sẵn kho
  PURCHASING: ["INBOUND_QC", "PLANNED", "CLOSED"], // PLANNED = PO cancel
  IN_PRODUCTION: ["PROD_QC", "PLANNED"], // V1.2 stub
  INBOUND_QC: ["AVAILABLE", "PLANNED"], // PLANNED = QC FAIL
  PROD_QC: ["AVAILABLE", "IN_PRODUCTION"], // V1.2 stub
  AVAILABLE: ["RESERVED", "CLOSED"],
  RESERVED: ["ISSUED", "AVAILABLE"], // AVAILABLE = unreserve
  ISSUED: ["ASSEMBLED"], // V1.2 stub
  ASSEMBLED: ["CLOSED"], // V1.2 stub
  CLOSED: [], // final
};

export function canTransition(from: BomSnapshotLineState, to: BomSnapshotLineState, isAdmin = false): boolean {
  if (from === "CLOSED") return false; // final, không admin override
  if (isAdmin) return BOM_SNAPSHOT_LINE_STATES.includes(to);
  return TRANSITION_MATRIX[from].includes(to);
}

export type TransitionGuard = {
  from: BomSnapshotLineState;
  to: BomSnapshotLineState;
  validate: (line: BomSnapshotLine, ctx: { userId: string; role: string; reason?: string }) => Promise<void>;
};

export const GUARDS: TransitionGuard[] = [
  // RESERVED → AVAILABLE: yêu cầu reason
  { from: "RESERVED", to: "AVAILABLE", validate: async (_, ctx) => {
    if (!ctx.reason || ctx.reason.length < 5) throw new HttpError(400, "Cần lý do ≥ 5 ký tự");
  }},
  // INBOUND_QC → AVAILABLE: yêu cầu qc_pass_qty > 0
  { from: "INBOUND_QC", to: "AVAILABLE", validate: async (line) => {
    if (Number(line.qcPassQty) <= 0) throw new HttpError(400, "Chưa có qty QC PASS");
  }},
  // INBOUND_QC → PLANNED: phải có QC FAIL decision
  // PLANNED → AVAILABLE: check item on-hand đủ không (V1.2 stub, chỉ warn)
];
```

```typescript
const transitionSchema = z.object({
  toState: z.enum(BOM_SNAPSHOT_LINE_STATES),
  versionLock: z.number().int().min(0),
  reason: z.string().optional(),
  context: z.record(z.any()).optional(),
});

export async function PATCH(req: Request, { params }: { params: { id: string } }) {
  const user = await requireAuth(req);
  const body = transitionSchema.parse(await req.json());
  
  return await db.transaction(async (tx) => {
    // 1. Load current line
    const [line] = await tx.select().from(bomSnapshotLine).where(eq(bomSnapshotLine.id, params.id));
    if (!line) throw new HttpError(404);
    
    const from = line.lineStatus;
    const isAdmin = user.role === "admin";
    
    // 2. Check transition allowed
    if (!canTransition(from, body.toState, isAdmin)) {
      throw new HttpError(400, `Không thể chuyển ${from} → ${body.toState}`);
    }
    
    // 3. Run guard
    const guard = GUARDS.find(g => g.from === from && g.to === body.toState);
    if (guard) await guard.validate(line, { userId: user.id, role: user.role, reason: body.reason });
    
    // 4. Optimistic lock UPDATE
    const result = await tx.update(bomSnapshotLine)
      .set({
        lineStatus: body.toState,
        versionLock: sql`version_lock + 1`,
        lastTransitionAt: new Date(),
        lastTransitionBy: user.id,
      })
      .where(and(
        eq(bomSnapshotLine.id, params.id),
        eq(bomSnapshotLine.lineStatus, from),
        eq(bomSnapshotLine.versionLock, body.versionLock),
      ))
      .returning();
    
    if (result.length === 0) {
      throw new HttpError(409, "State đã bị thay đổi bởi user khác, refresh lại");
    }
    
    // 5. Log transition
    await tx.insert(stateTransitionLog).values({
      entityType: "bom_snapshot_line",
      entityId: params.id,
      fromStatus: from,
      toStatus: body.toState,
      actorUserId: user.id,
      actorRole: user.role,
      reason: body.reason,
      adminOverride: isAdmin && !TRANSITION_MATRIX[from].includes(body.toState),
      contextJson: body.context ?? {},
    });
    
    // 6. Audit
    await tx.insert(auditLog).values({
      action: "TRANSITION", entityType: "bom_snapshot_line", entityId: params.id,
      userId: user.id, context: { from, to: body.toState, reason: body.reason },
    });
    
    // 7. Invalidate dashboard
    await redis.del("dashboard:overview");
    
    return Response.json(result[0]);
  });
}
```

```typescript
async function convertPRToPO(prId: string, userId: string) {
  return await db.transaction(async (tx) => {
    const pr = await getPR(prId);
    if (pr.status !== "APPROVED") throw new HttpError(400, "PR chưa APPROVED");
    
    const lines = await getPRLines(prId);
    
    // Validate tất cả line có preferredSupplierId
    const missing = lines.filter(l => !l.preferredSupplierId);
    if (missing.length) throw new HttpError(400, `${missing.length} line thiếu NCC: ${missing.map(l => l.itemId).join(",")}`);
    
    // Group by supplier
    const groups = new Map<string, PRLine[]>();
    for (const l of lines) {
      const arr = groups.get(l.preferredSupplierId!) ?? [];
      arr.push(l); groups.set(l.preferredSupplierId!, arr);
    }
    
    const createdPOs = [];
    for (const [supplierId, grpLines] of groups) {
      const poNo = `PO-${format(new Date(), "yyMM")}-${nanoid(4).toUpperCase()}`;
      const [po] = await tx.insert(purchaseOrder).values({
        poNo, supplierId, linkedOrderId: pr.linkedOrderId, linkedPrId: prId,
        status: "DRAFT", createdBy: userId,
      }).returning();
      
      for (const [idx, l] of grpLines.entries()) {
        await tx.insert(purchaseOrderLine).values({
          poId: po.id, lineNo: idx + 1, itemId: l.itemId,
          orderedQty: l.requestedQty, expectedEta: l.neededBy,
          linkedSnapshotLineId: l.linkedSnapshotLineId,
        });
        
        // Side effect: snapshot line update
        if (l.linkedSnapshotLineId) {
          await tx.update(bomSnapshotLine)
            .set({ openPurchaseQty: sql`open_purchase_qty + ${Number(l.requestedQty)}` })
            .where(eq(bomSnapshotLine.id, l.linkedSnapshotLineId));
          
          // Transition PLANNED → PURCHASING
          const [line] = await tx.select().from(bomSnapshotLine).where(eq(bomSnapshotLine.id, l.linkedSnapshotLineId));
          if (line?.lineStatus === "PLANNED") {
            await tx.update(bomSnapshotLine)
              .set({ lineStatus: "PURCHASING", versionLock: sql`version_lock + 1`, lastTransitionAt: new Date(), lastTransitionBy: userId })
              .where(eq(bomSnapshotLine.id, l.linkedSnapshotLineId));
            await tx.insert(stateTransitionLog).values({
              entityType: "bom_snapshot_line", entityId: l.linkedSnapshotLineId,
              fromStatus: "PLANNED", toStatus: "PURCHASING", actorUserId: userId,
              reason: `Auto-transition khi convert PR ${pr.prNo}`, contextJson: { prId, poId: po.id },
            });
          }
        }
      }
      
      createdPOs.push(po);
    }
    
    // Update PR status
    await tx.update(purchaseRequest).set({ status: "CONVERTED" }).where(eq(purchaseRequest.id, prId));
    
    // Audit
    await tx.insert(auditLog).values({ action: "CONVERT_PR_PO", entityType: "purchase_request", entityId: prId, userId, context: { poIds: createdPOs.map(p => p.id) } });
    
    return createdPOs;
  });
}
```

```typescript
      action: "RECEIVE", entityType: "receiving_event", entityId: event.id, userId: user.id,
      context: { poCode: body.poCode, sku: body.sku, qty: body.qty, lotNo: body.lotNo },
    });
    
    // 13. Warning over-delivery
    const overDelivery = Number(poLine.receivedQty) + Number(body.qty) > Number(poLine.orderedQty) * 1.05;
    
    // 14. Invalidate caches
    await redis.del("dashboard:overview");
    
    return Response.json({ ...event, warning: overDelivery ? "Qty nhận > 105% ordered" : null });
  });
}
```

**QC flow API:** `apps/web/src/app/api/receiving/qc/[receiptId]/route.ts` POST (~120 LOC)

```typescript
const qcSchema = z.object({ decision: z.enum(["PASS", "FAIL"]), notes: z.string().optional() });

export async function POST(req, { params }) {
  const user = await requireAuth(req, ["admin", "planner", "operator"]);
  const { decision, notes } = qcSchema.parse(await req.json());
  
  return await db.transaction(async (tx) => {
    const [receipt] = await tx.select().from(inboundReceipt).where(eq(inboundReceipt.id, params.receiptId));
    if (!receipt) throw new HttpError(404);
    if (receipt.qcFlag !== "PENDING") throw new HttpError(400, "Receipt đã QC");
    
    // Update receipt
    await tx.update(inboundReceipt).set({
      qcFlag: decision, qcCheckedAt: new Date(), qcCheckedBy: user.id, qcNotes: notes, isPosted: true,
    }).where(eq(inboundReceipt.id, params.receiptId));
    
    // Get all lines
    const lines = await tx.select().from(inboundReceiptLine).where(eq(inboundReceiptLine.receiptId, params.receiptId));
    
    for (const rl of lines) {
      // Find PO line → snapshot line
      const [poLine] = await tx.select().from(purchaseOrderLine).where(eq(purchaseOrderLine.id, rl.poLineId));
      if (!poLine?.linkedSnapshotLineId) continue;
      
      const [snapLine] = await tx.select().from(bomSnapshotLine).where(eq(bomSnapshotLine.id, poLine.linkedSnapshotLineId));
      if (!snapLine) continue;
      
      if (decision === "PASS") {
        // Update qc_pass_qty + transition INBOUND_QC → AVAILABLE
        await tx.update(bomSnapshotLine)
          .set({
            qcPassQty: sql`qc_pass_qty + ${Number(rl.receivedQty)}`,
            lineStatus: "AVAILABLE",
            versionLock: sql`version_lock + 1`,
            lastTransitionAt: new Date(),
            lastTransitionBy: user.id,
          })
          .where(eq(bomSnapshotLine.id, snapLine.id));
        await tx.insert(stateTransitionLog).values({
          entityType: "bom_snapshot_line", entityId: snapLine.id,
          fromStatus: snapLine.lineStatus, toStatus: "AVAILABLE", actorUserId: user.id,
          reason: `QC PASS receipt ${receipt.receiptNo}`, contextJson: { receiptId: params.receiptId, rlId: rl.id },
        });
      } else {
        // FAIL: lot HOLD + transition INBOUND_QC → PLANNED
        if (rl.lotCode) {
          await tx.update(inventoryLotSerial)
            .set({ status: "HOLD", notes: `QC FAIL ${receipt.receiptNo}: ${notes ?? ""}` })
            .where(and(eq(inventoryLotSerial.itemId, rl.itemId), eq(inventoryLotSerial.lotCode, rl.lotCode)));
        }
        await tx.update(bomSnapshotLine)
          .set({
            lineStatus: "PLANNED",
            versionLock: sql`version_lock + 1`,
            lastTransitionAt: new Date(),
            lastTransitionBy: user.id,
          })
          .where(eq(bomSnapshotLine.id, snapLine.id));
        await tx.insert(stateTransitionLog).values({
          entityType: "bom_snapshot_line", entityId: snapLine.id,
          fromStatus: snapLine.lineStatus, toStatus: "PLANNED", actorUserId: user.id,
          reason: `QC FAIL receipt ${receipt.receiptNo}: ${notes}`, contextJson: { receiptId: params.receiptId },
        });
      }
    }
    
    await tx.insert(auditLog).values({
      action: "APPROVE", entityType: "inbound_receipt", entityId: params.receiptId,
      userId: user.id, context: { decision, notes, lineCount: lines.length },
    });
    
    await redis.del("dashboard:overview");
    
    return Response.json({ receiptId: params.receiptId, decision, affectedSnapshotLines: lines.length });
  });
}
```
